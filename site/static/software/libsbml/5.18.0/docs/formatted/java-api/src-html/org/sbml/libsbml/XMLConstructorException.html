001package org.sbml.libsbml;
002
003/**
004 * Exceptions class for libSBML object constructors.
005 * <p>
006 * <em style='color: #555'>
007 * This class of objects is defined by libSBML only and has no direct
008 * equivalent in terms of SBML components.
009 * </em>
010 * <p>
011 * This class makes it possible to catch exceptions thrown by some libSBML
012 * constructors in the XML interface layer.  Callers can use the class
013 * in code such as the following example:
014 * <div class='fragment'><pre>
015XMLAttributes x;
016<em>... do things with XMLAttributes object x ...</em>
017
018XMLAttributes xcopy;
019try
020{
021    xcopy = new XMLAttributes(x);
022}
023catch (XMLConstructorException e)
024{
025    String errmsg = e.getMessage();
026}
027</pre></div>
028 * <p>
029 * Not all libSBML object classes throw this exception; the cases are
030 * limited to the classes that create SBML objects (e.g.,
031 * {@link Compartment}, {@link Species}, etc.) and {@link SBMLDocument}.
032 */
033public class XMLConstructorException extends java.lang.IllegalArgumentException
034{
035    /**
036     * Constructor; version taking a message argument.
037     * <p>
038     * @param message The exception message.
039     */
040    public XMLConstructorException(String message) {}
041
042
043    /**
044     * Constructor; version taking no arguments.
045     */
046    public XMLConstructorException() {}
047
048
049    /**
050     * Destructor for this list.
051     * <p>
052     * Callers can use this method to delete this object explicitly
053     * after it is no longer needed.
054     */
055    public synchronized void delete() { }
056}
