001package org.sbml.libsbml;

003/**
004 * <span class='pkg-marker pkg-color-layout'><a href='group__layout.html'>layout</a></span>
005 *
006 * Class to store SBML Level, Version, and XML namespace for the
007 * SBML Layout (&ldquo;layout&rdquo;) package.
008 * <p>
009 * <em style='color: #555'>
010 * This class of objects is defined by libSBML only and has no direct
011 * equivalent in terms of SBML components.  This class is not prescribed by
012 * the SBML specifications, although it is used to implement features
013 * defined in SBML.
014 * </em>
015 * <p>
016 * SBML Level&nbsp;3 &ldquo;packages&rdquo; add features on top of SBML
017 * Level&nbsp;3 Core.  When a model definition uses an SBML package, it must
018 * declare the Level and Version of SBML Core and the Version of the package
019 * being used.  The package name, the SBML Level and Version, and the package
020 * Version correspond uniquely to an XML namespace added to the XML encoding
021 * of the SBML model.
022 * <p>
023 * LibSBML Level&nbsp;3 <em>extensions</em> are implementations of support
024 * for SBML Level&nbsp;3 packages.  Each package is implemented as a separate
025 * extension.  To allow software applications to query the level and version
026 * information of an extension's package implementation, libSBML uses
027 * specialized object classes.  For the extension implementing the SBML
028 * &ldquo;layout&rdquo; package, the object class is {@link
029 * LayoutPkgNamespaces}.  (This class is a specialization of a common base
030 * class called <code>SBMLExtensionNamespaces</code> that is not exposed in
031 * the libSBML programming language interfaces other than C++.)
032 * <p>
033 * Objects of class {@link LayoutPkgNamespaces} can be passed to constructors
034 * of SBML components defined by &ldquo;layout&rdquo; in order to ensure that
035 * the correct component structure is created.  This is necessary because
036 * different versions of an SBML Level&nbsp;3 package may introduce
037 * differences in the definition of the components defined by the package.
038 * (For example, later editions of a package may introduce new attributes on
039 * a component that are not present in earlier editions of the package
040 * specification.)
041 * <p>
042 * @see FbcPkgNamespaces
043 * @see LayoutPkgNamespaces
044 * @see QualPkgNamespaces
045 */
046public class LayoutPkgNamespaces {
047
048    /**
049     * Creates a new {@link LayoutPkgNamespaces} object.
050     * <p>
051     * @warning Note that the internal implementation of the list nodes uses
052     * C++ objects.  If callers use this constructor to create the list
053     * object deliberately, those objects are in a sense "owned" by the caller
054     * when this constructor is used. Callers need to remember to call
055     * {@link #delete()} on this list object after it is no longer
056     * needed or risk leaking memory.
057     */
058    public LayoutPkgNamespaces() { }
059
060
061    /**
062     * Creates and returns a deep copy of this {@link LayoutPkgNamespaces}.
063     * <p>
064     * @return a (deep) copy of this {@link LayoutPkgNamespaces}.
065     */ 
066    public LayoutPkgNamespaces cloneObject() {}
067
068
069    /**
070     * Destroys this object.
071     * <p>
072     * If a caller created this list using the {@link #LayoutPkgNamespaces()}
073     * constructor, the caller should use this method to delete this list
074     * object after it is no longer in use.
075     */
076    public synchronized void delete() { }
077
078
079    /**
080     * Returns a string representing the package's XML namespace.
081     * <p>
082     * @return a string representing the XML namespace that reflects the
083     * SBML Level and Version, and the package Version.
084     */
085    public String getURI() { }
086
087
088    /**
089     * Returns the package's version number.
090     * <p>
091     * @return an integer, the version number for the SBML Level&nbsp;3
092     * Layout package implemented by the libSBML
093     * extension.
094     */
095    public long getPackageVersion() { }
096
097
098    /**
099     * Returns the short-form name or label for the package.
100     * <p>
101     * @return the short-form name of the SBML Level&nbsp;3 package, which in
102     * this case, will be &ldquo;layout&rdquo;.
103     */ 
104     public String getPackageName() { }
105}
