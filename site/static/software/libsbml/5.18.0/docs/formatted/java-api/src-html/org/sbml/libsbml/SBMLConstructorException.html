001package org.sbml.libsbml;
002
003/**
004 * Exceptions class for libSBML object constructors.
005 * <p>
006 * <em style='color: #555'>
007 * This class of objects is defined by libSBML only and has no direct
008 * equivalent in terms of SBML components.
009 * </em>
010 * <p>
011 * This class makes it possible to catch exceptions thrown by many libSBML
012 * constructors.  Callers can use the class in code such as the
013 * following example:
014 * <div class='fragment'><pre>
015Model m;
016try
017{
018    m = new Model(level, version);
019}
020catch (SBMLConstructorException e)
021{
022    String errmsg = e.getMessage();
023}
024</pre></div>
025 * <p>
026 * Not all libSBML object classes throw this exception; the cases are
027 * limited to the classes that create SBML objects (e.g.,
028 * {@link Compartment}, {@link Species}, etc.) and {@link SBMLDocument}.
029 */
030public class SBMLConstructorException extends java.lang.IllegalArgumentException
031{
032    /**
033     * Constructor; version taking a message argument.
034     * <p>
035     * @param message The exception message.
036     */
037    public SBMLConstructorException(String message) {}
038
039
040    /**
041     * Constructor; version taking no arguments.
042     */
043    public SBMLConstructorException() {}
044
045
046    /**
047     * Destructor for this list.
048     * <p>
049     * Callers can use this method to delete this object explicitly
050     * after it is no longer needed.
051     */
052    public synchronized void delete() { }
053}
