001package org.sbml.libsbml;
002
003/**
004 * Class for managing lists of {@link SBase} objects.
005 * <p>
006 * <em style='color: #555'>
007 * This class of objects is defined by libSBML only and has no direct
008 * equivalent in terms of SBML components.
009 * </em>
010 * <p>
011 * This class is necessary because of programming language differences
012 * between Java and the underlying C++ core of libSBML's implementation.
013 * It would of course be preferable to have a common list type for all
014 * lists returned by libSBML (e.g., lists of {@link SBase} objects, lists
015 * of {@link CVTerm} objects, etc.).  However, this is currently impossible
016 * to achieve given the way the underlying C++ lists are implemented.  (The
017 * basic problem concerns the lack of an equivalent to <code>void *</code>
018 * pointers in Java.)
019 * <p>
020 * As a result of this incompatibility, libSBML must implement the Java
021 * versions of the lists in another way.  The approach taken is to
022 * define specialized list types for each kind of object that needs
023 * a list; that is, {@link SBaseList} for {@link SBase} objects,
024 * {@link CVTermList} for {@link CVTerm} objects, and a few others.
025 * These list objects provide the same kind of functionality that
026 * the underlying C++ generic lists provide (such as <code>get()</code>,
027 * <code>add()</code>, <code>remove()</code>, etc.), yet still
028 * maintain the strong data typing requiring by Java.
029 */
030public class SBaseList {
031
032    /**
033     * Explicit constructor for this list.
034     * <p>
035     * In most circumstances, callers will obtain an {@link SBaseList}
036     * object from a call to a libSBML method that returns the list.
037     * However, the constructor is provided in case callers need to
038     * construct the lists themselves.
039     * <p>
040     * @warning Note that the internal implementation of the list nodes uses
041     * C++ objects.  If callers use this constructor to create the list
042     * object deliberately, those objects are in a sense "owned" by the caller
043     * when this constructor is used. Callers need to remember to call
044     * {@link #delete()} on this list object after it is no longer
045     * needed or risk leaking memory.
046     */
047    public SBaseList() { }
048
049
050    /**
051     * Destructor for this list.
052     * <p>
053     * If a caller created this list using the {@link #SBaseList()}
054     * constructor, the caller should use this method to delete this list
055     * object after it is no longer in use.
056     */
057    public synchronized void delete() { }
058
059
060    /**
061     * Adds the given {@link SBase} object <code>item</code> to this
062     * list.
063     * <p>
064     * @param item the {@link SBase} object to add to add
065     */
066    public void add(SBase item) { }
067
068
069    /**
070     * Returns the <em>n</em>th SBase object from this list.
071     * <p>
072     * If the index number <code>n</code> is greater than the size of the list
073     * (as indicated by {@link #getSize()}), then this method returns
074     * <code>null</code>.
075     * <p>
076     * @param n the index number of the item to get, with indexing
077     * beginning at number <code>0</code>.
078     * <p>
079     * @return the nth item in this {@link SBaseList} items.
080     * <p>
081     * @see #getSize()
082     */
083    public SBase get(long n) { }
084
085
086    /**
087     * Adds the {@link SBase} object <code>item</code> to the beginning
088     * of this list.
089     * <p>
090     * @param item a pointer to the item to be prepended.
091     * <p>
092     */
093    public void prepend(SBase item) { }
094
095
096    /**
097     * Removes the <em>n</em>th {@link SBase} object from this list and
098     * returns it.
099     * <p>
100     * Callers can use {@link #getSize()} to find out the length of the list.
101     * If <code>n > </code>{@link #getSize()}, this method returns
102     * <code>null</code> and does not delete anything.
103     * <p>
104     * @param n the index number of the item to remove
105     * <p>
106     * @return the item indexed by <code>n</code>
107     * </p>
108     * @see #getSize()
109     */
110    public SBase remove(long n) { }
111
112
113    /**
114     * Returns the number of items in this list.
115     * <p>
116     * @return the number of elements in this list.
117     */
118    public long getSize() { }
119
120}
